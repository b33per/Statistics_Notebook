# Skills Quiz - Different Types of Models (Problem 4)

set.seed(101) 

n <- 100 

X_i <- runif(n, -2, 2)


beta0.1 <- -2 
beta1.1 <- 3
beta2.1 <- 4

beta0.2 <- 2
beta1.2 <- 5
beta2.2 <- -3

sigma <- 2

epsilon_i <- rnorm(n, 1, sigma)  


Y_i.1 <- beta0.1 + beta1.1*X_i + beta2.1*X_i^2 + epsilon_i
Y_i.2 <- beta0.2 + beta1.2*X_i + beta2.2*X_i^2 + epsilon_i
